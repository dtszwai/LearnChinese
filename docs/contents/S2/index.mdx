```mdx-code-block
import PassageList from "./PassageList.json";
import Link from '@docusaurus/Link';

export const RenderList = () =>
  PassageList.filter((item) => typeof item === "object").map((item, i) => (
    <div className="text-list" style={{ color: "#82735C" }} key={i}>
      <p>
        <Link to={`/contents/S2/${item.label.split("：").at(-1)}`}>{item.label}</Link>
      </p>
      <ul>
        {item.items.map((passage, j) => (
          <li key={j}>
            <Link to={`/${passage}`}>{passage.split("/")[1]}</Link>
          </li>
        ))}
      </ul>
    </div>
  ));

<RenderList />
```
